原型继承

//组合继承
function Parent(value){
    this.val = value;
}
Parent.prototype.getValue = function(){
    console.log(this.val)
}
function Child(value){
    Parent.call(this,value)
}
Child.prototype = new Parent()
//Child.prototype.constructor = Child;

let child = new Child(1)
child.getValue() //1 
child instanceof Parent //true

子类构造函数通过Parent.call(this,value) 继承父类的属性， 然后子类原型指向父类的函数, 但是缺点是 在继承父类函数调用了父类构造函数，会导致子类原型多了不需要父类属性，导致内存浪费。

寄生组合继承,优化在继承父类调用了构造函数
function Parent(value){
    this.val = value
}
Parent.prototype.getValue = function(){
    console.log(this.val)
}
function Child(value){
    Parent.call(this,value)
}
Child.prototype = Object.create(Parent.prototype,{
    constructor:{
        value: Child,
        enumerable: false,
        writable: true,
        configurable: true
    }
})
let child = new Child(1)

父类的原型赋值给了子类，构造函数设置为子类，解决了父类属性问题，也解决了子类构造函数

Function.prototype 本身没有valueOf,是继承了Object.prototype对象的valueOf
Function.prototype 又和其他函数的prototype 不一样，只有函数有prototype，而它没有，理解的是引擎创建的函数，不然Function.prototype.prototype.... 没有意义

Object作为构造函数的时候，其内部属性[[prototype]]指向Function.prototype
Object.__proto__ === Function.prototype

Function构造函数,Function 的 [[Prototype]] 属性指向了 Function.prototype

在此处， Function 对象是不是由 Function 构造函数创建的一个实例？

__proto__是浏览器内部实现，而new出来的实例proto指向构造函数的prototype，这就是new做的事情

寄生组合继承: 1.子类构造函数的proto指向父类构造器，继承父类静态方法 2.子类构造函数的prototype的proto指向父类构造器的prototype，继承父类的方法 3.子类构造器里调用父类构造器，继承父类的属性

new 实现

1.创建一个新对象 new的结果是一个新对象 2.这个对象会被执行[prototype]链接 3.生成的新对象绑定函数调用的this 4.通过new创建每个对象被[[prototype]]链接到这个函数prototype对象上 5.函数没有返回对象Object，new表达式中函数调用 会自动返回这个新对象

function objectFactory() {
    var obj = new Object()
    Constructor = [].shift.call(arguments) //取出第一个构造函数，shift会修改原数组，arguments 会被去除第一个参数
    obj.__proto__ = Constructor.prototype //新对象访问原型的属性
    Constructor.apply(obj,arguments)     //新对象访问构造函数的属性
    return obj
}

bind实现


Function.prototype.bind = Function.prototype.bind || function bind(thisArg){
    if(typeof this !== 'function'){
        throw new TypeError(this + ' must be a function');
    }
    var self = this;
    // 去除thisArg的其他参数 转成数组
    var args = [].slice.call(arguments, 1);
    var bound = function(){
         // bind返回的函数 的参数转成数组
        var boundArgs = [].slice.call(arguments);
        var finalArgs = args.concat(boundArgs);
        // new 调用时，其实this instanceof bound判断也不是很准确。es6 new.target就是解决这一问题的。
        if(this instanceof bound){
            // 1.创建一个全新的对象
            // 2.这个对象会被执行[[prototype]](proto)链接
            // 4.通过`new`创建的每个对象将最终被`[[Prototype]]`链接到这个函数的`prototype`对象上。
            // self可能是ES6的箭头函数，没有prototype，所以就没必要再指向做prototype操作。
            if(self.prototype){
                // ES5 提供的方案 Object.create()
                // bound.prototype = Object.create(self.prototype);
                // 但 既然是模拟ES5的bind，那浏览器也基本没有实现Object.create()
                // 所以采用 MDN ployfill方案 https://developer.mozilla.org/zh-C
                function Empty(){}
                Empty.prototype = self.prototype;
                bound.prototype = new Empty();
            }
            //3.生成的新对象绑定函数调用的`this`
            var result = self.apply(this, finalArgs);
             // 5.如果函数没有返回对象类型`Object`(包含`Functoin`, `Array`, `Date`, `RegExg`, `Error`)，
            // 那么`new`表达式中的函数调用会自动返回这个新的对象。
            var isObject = typeof result === 'object' && result !== null;
            var isFunction = typeof result === 'function';
            if(isObject || isFunction){
                return result;
            }
            return this;
        }
        else{
            // apply修改this指向，把两个函数的参数合并传给self函数，并执行self函数，返回执行结果
            return self.apply(thisArg, finalArgs);
        }
    };
    return bound;
}

apply实现

Function.prototype.apply = function (context, arr) {
    var context = Object(context) || window;
    context.fn = this;

    var result;
    if (!arr) {
        result = context.fn();
    }
    else {
        var args = [];
        for (var i = 0, len = arr.length; i < len; i++) {
            args.push('arr[' + i + ']');
        }
        result = eval('context.fn(' + args + ')')
    }

    delete context.fn
    return result;
}

Function.protype.bind2 = function (ctx) {
    var self = this;
    var args = Array.protype.slice.call(argument,1)
    var bindFn = function(){
        var bindArgs = Array.prototype.slice.call(arguments);
        //返回函数作为构造函数时，this指向实例，将绑定函数this指向实例，让实例获得来自绑定的值
        //普通函数时，this->window 绑定函数this指向 ctx
        return self.apply(this instanceof bindFn ? this :ctx,args.concat(bindArgs))
    }
    //返回函数的prototype 为绑定函数的prototype，实例继承绑定函数原型的值
    //此处prototype修改，会导致直接修改绑定函数的prototype
    bindFn.prototype = this.prototype

    return bindFn
}

//修改版
Function.protype.bind2 = function (ctx) {
    if (typeof this !== "function") {
        throw new Error("Function.prototype.bind - what is trying to be bound is not callable");
    }
    var self = this;
    var args = Array.protype.slice.call(argument,1)
    var temFn = function(){}
    var bindFn = function(){
        var bindArgs = Array.prototype.slice.call(arguments);
        //返回函数作为构造函数时，this指向实例，将绑定函数this指向实例，让实例获得来自绑定的值
        //普通函数时，this->window 绑定函数this指向 ctx
        return self.apply(this instanceof temFn ? this :ctx,args.concat(bindArgs))

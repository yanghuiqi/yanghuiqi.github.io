迭代器 for of

迭代器

    // value done 每次next 看是否value 
    // return throw 方法是可选 return方法是for of 提前退出（break,continue) 调用return
    function createIterator(items) {
        var i = 0;
        return {
            next: function() {
                var done = i >= items.length;
                var value = !done ? items[i++] : undefined;

                return {
                    done: done,
                    value: value
                };
            },
            return: function() {4
                console.log("执行了 return 方法");
                return {
                    value: 23333,
                    done: true
                };
            }
        };
    }
    var colors = ["red", "green", "blue"];

    var iterator = createIterator([1, 2, 3]);

    colors[Symbol.iterator] = function() {
        return iterator;
    };

    for (let color of colors) {
        if (color == 1) break;
        console.log(color);
    }

既然是迭代，就必须有迭代条件， 一个数据结构有 Symbol.iterator 就可以认为可遍历

实现for of

通过Symbol.iterator属性获取迭代器对象

    function forOf(obj,cb){
        let iterable,result

        iterable = obj[Symbol.iterator]()
        result = iterable.next()
        while(!result.done){
            cb(result.value)
            iterable = iterable.next()
        }
    }

遍历方法都是遍历器
1.kyes() 
2.value()
3.entries() 
4.forEach() 回调函数遍历每个成员

   (function(global){
       var NaNSymbol = Symbol('NaN')
       var encodeVal = function(value){
           return value !==value ? NaNSymbol : value
       }
       function Set(data){
           this._values = []
           this.size = 0
           data && data.forEach(function(item){
               this.add(item);
           },this)
       }
       Set.prototype['add'] = function(value){
           value = encodeVal(value)
           if(this._values.indexOf(value) == -1){
               this._values.push(value)
               ++this.size
           }
           return this
       }
       Set.prototype['has'] = function(value){
           return (this._values.indexOf(encodeVal(val)) !== -1)
       }
       Set.prototype['delete'] = function(value){
           var idx = this._values.indexOf(encodeVal(value))
           if(idx == -1) return false
           this._values.splice(idx,1)
           --this.size
           return this
       }
       Set.prototype['clear'] = function(value){
           this._values = []
           this.size = 0
       }
       Set.prototype['forEach'] = function(callbackFn,thisArg){
           for(var i = 0; i < this._values.length; i++){
               callbackFn.call(thisArg,this._values[i],this._values[i],this)
           }
       }
       Set.length = 0
       global.Set = Set
   })(this)

WeakMap

特性

只接受对象作为键值名

    const map = new WeakMap()
    map.set(1,2)//Invalid value used as weak map key

键名所引用的对象都是弱引用

TIP

WeakMap 保持了对键名所引用对象的弱引用，也就说一个对象被弱引用，会随时被回收

在js中，创建的对象都是强引用

    var obj = new Object()
    //手动obj = null 才会回收，

    var obj = new WeakMap()
    // obj 等待回收即可

    var key = Array(5).fill(2) 
    var arr = [
        [key,1]
    ]

如上，arr对key所引用对象的强引用，key = null只是去掉了key对Obj的强引用，而arr对key强引用没去掉，Obj还是不会被回收

    let map =  new Map()
    let key = Array(5).fill(2) 
    map.set(key,1)
    map.delete(key)
    key = null

如果才会被回收

    let obj = new WeakMap()
    let key = Array(5).fill(2) 
    obj.set(key,1)
    key = null

如上，建立了obj对key所引用对象的弱引用，key对所引用对象的强引用，不会被强制回收，但是置为null，就被GC(回收对象所占内存)了，而Obj只是建立了弱引用，该引用对象会被GC，也就是说，不需要的时候，不需要WeakMap 手动删除，会被自动GC，故是可变的，没有遍历方法 只要 get set has delete

解决回调地狱 而回调地狱，不只是书写繁琐，还有几个方面需要注意: 1.复用性较差 2.调用堆栈出现错误 当一个函数执行的时候，js引擎创建该函数执行上下文入栈，执行完毕，出栈，当是一个回调函数入队之后，主线程执行完毕，才会执行任务队列，将其加入栈中，而此刻上下文只有该回调，try catch无法捕获错误（异步） 3.借助第三变量 不易维护，无法将第三变量状态正确修改，而且当出现错误，还会继续回调，造成多余浪费，而且也容易被同一作用域修改，造成不必要的bug //实现一个PromiseA+

    function my_co(it) {
        return new Promise((resolve, reject) => {
            function next(data) {
                try {
                    var { value, done } = it.next(data);
                }catch(e){
                    return reject(e);
                }
                if (!done) { 
                    //done为true,表示迭代完成
                    //value 不一定是 Promise，可能是一个普通值。使用 Promise.resolve 进行包装。
                    Promise.resolve(value).then(val => {
                        next(val);
                    }, reject);
                } else {
                    resolve(value);
                }
            }
            next(); //执行一次next
        });
    }

用法

    Object.defineProperty(obj,props,descriptor)
    // obj : 定义属性的对象
    //props: 定义修改属性的名称
    //descriptor ： 定义或修改属性的描述符

    var obj = {}
     Object.defineProperty(obj,"a",{
         value:1,
         writable:true,
         enumerable:true,
         configurable:true
     })

关于第三个参数descriptor表示属性描述符2种形式:数据描述符和存取描述符

共同键值:

configurable 为true 表示属性可以被改变，或删除

enumerable 为true表示属性可以出现对象枚举属性中

数据描述符键值:

value 属性对应的值 默认为undefined

writable 为true赋值运算符可改变 默认false

存取描述符键值:

get 提供了getter方法，返回值用作属性值

set 提供了setter方法，该方法提供唯一参数，用作新值分配给该属性 没有则为undefined

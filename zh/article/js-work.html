函数防抖

原理:触发事件是在触发N秒后执行，如果在一个限定事件内触发，就会清除这个事件，直到触发事件N秒后不再执行，才执行。

function debounce(func,wait){
    var timeout;
    return function (){
        var ctx =this //this指向
        var args = arguments //参数保存 (event对象)
        clearTimeout(timeout)
        timeout = setTimeout(function(){
            func.apply(ctx,args)
        },wait)
    }
}

立即执行

我希望第一次就执行，然后触发N秒后，再次触发第二次事件

function debounce(func,wait,immediate){
    var timeout,result;
    return function (){
        var ctx =this //this指向
        var args = arguments //参数保存 (event对象)
        timeout && clearTimeout(timeout)
        if(immediate){
            //已执行，就不再执行
            var callRun = !timeout;
            timeout = setTimeout(function(){
                timeout = null
            },wait)
            //callRun为true 才会执行第一次或者第二次符合等待N秒的事件
            callRun && (result = func.apply(ctx,args))
        }else {
            timeout = setTimeout(function(){
                func.apply(ctx,args)
            },wait)
        }
        return result;
    }
}

取消防抖

比如一个场景，5秒内执行首次，紧接着等待5s再次执行，那么可否点击立即执行，继续下次的防抖呢？

function debounce(func,wait,immediate){
    var timeout,result;
    var deFn = function (){
        var ctx =this //this指向
        var args = arguments //参数保存 (event对象)
        timeout && clearTimeout(timeout)
        if(immediate){
            //已执行，就不再执行
            var callRun = !timeout;
            timeout = setTimeout(function(){
                timeout = null
            },wait)
            //callRun为true 才会执行第一次或者第二次符合等待N秒的事件
            callRun && (result = func.apply(ctx,args))
            else {
                timeout = setTimeout(function(){
                    func.apply(ctx,args)
                },wait)
            }
        }
        return result;
    }
    deFn.cancel = function(){
        //清除定时器并且timeout为null，callRun便会为true
        clearTimeout(timeout)
        timeout = null
    }
    return deFn
}

函数节流

原理:每隔一段时间执行一次，可以类比为水龙头滴水

时间戳实现

当前时间-上次时间 > 设定时间 ， 函数执行

function throttle (func,wait){
    var ctx,args,
        prev = 0;
    return function () {
        ctx = this
        args = arguments
        var now = +new Date();
        if(now-prev >wait){
            func.apply(ctx,args)
            prev = now
        }
    }
}

定时器实现

设置一个定时器，定时器不存在，就不执行，当执行完清空定时器

function throttle (func,wait){
    var timeout,
        ctx,
        args
    return function(){
        ctx = this
        args = arguments
        if(!timeout){
            timeout = setTimeout(function(){
                timeout = null
                func.apply(ctx,args)
            },wait)
        }
    }
}

总结:
1.第一个会立即执行，第二个会N秒执行
2.第一个事件停止触发不再执行，第二个相反。

时间和计时器两者结合的节流（立即执行，停止触发后再执行一次）

function throttle (func,wait){
    var timeout,args,prev = 0,ctx,result;
    var later = function(){
        prev = +new Date()
        timeout = null
        func.apply(ctx,args)
    }
    var throFn = function(){
        var now = +new Date()
        var remaining = wait - (now - prev)
        ctx = this
        args = arguments
        if(remaining <= 0 || remaining >wait){
            if(timeout){
                clearTimeout(timeout)
                timeout = null
            }
            prev = now
            func.apply(ctx,args)
        }else if (!timeout){
            timeout = setTimeout(later,remaining)
        }
    }
    return throFn
}

function isPlainObject(obj){
    var proto,Ctor
    //排除 window obj对象
    if(!obj || toString.call(obj) !== "[object Object]"){
        return false
    }
    //获取 原型  new Object => obj.__proto__ === Object.prototype
    proto = obj.getPrototypeOf(obj)
    // Object.create(null) 返回true 因为里面什么都没有
    if(!proto){
        return true
    }
    //new Object创建对象，是否有构造器 && Ctor 就为proto的构造器
    // Object.create(null) 创建的对象那个就等于 Object的构造函数
    Ctor = hasOwn.call(proto,"constructor") && proto.constructor
    //Ctor的构造函数 是不是Object构造函数
    //这里是hasOwnPrototype.toString.call  => Function.prototype.toString 
    //Function 对象覆盖了从 Object 继承来的 Object.prototype.toString 方法。函数的 toString 方法会返回一个表示函数源代码的字符串。具体来说，包括 function关键字，形参列表，大括号，以及函数体中的内容。
    return typeof Ctor ==== "function" && hasOwn.toString.call(Ctor) === hasOwn.toString.call(Object)
}

function Find(target, array) {
    let i = 0;
    let j = array[i].length - 1;
    while (i < array.length && j >= 0) {
        if (array[i][j] < target) {
            i++;
        } else if (array[i][j] > target) {
            j--;
        } else {
            return true;
        }
    }
    return false;
}

判断两个对象是否相等

+0 -0

JavaScript 采用了IEEE_754 浮点数表示法(几乎所有现代编程语言所采用)，这是一种二进制表示法，按照这个标准，最高位是符号位(0 代表正，1 代表负)，剩下的用于表示大小。而对于零这个边界值 ，1000(-0) 和 0000(0)都是表示 0 ，这才有了正负零的区别。

    1 / +0 // Infinity
    1 / -0 // -Infinity

    1 / +0 === 1 / -0 // false
    function eq(a, b){
        if (a === b) return a !== 0 || 1 / a === 1 / b;
        return false;
    }
    console.log(eq(0, 0)) // true
    console.log(eq(0, -0)) // false

    // eq 第一版
    // 用来过滤掉简单的类型比较，复杂的对象使用 deepEq 函数进行处理
    function eq(a, b) {

        // === 结果为 true 的区别出 +0 和 -0
        if (a === b) return a !== 0 || 1 / a === 1 / b;

        // typeof null 的结果为 object ，这里做判断，是为了让有 null 的情况尽早退出函数
        if (a == null || b == null) return false;

        // 判断 NaN
        if (a !== a) return b !== b;

        // 判断参数 a 类型，如果是基本类型，在这里可以直接返回 false
        var type = typeof a;
        if (type !== 'function' && type !== 'object' && typeof b != 'object') return false;

        // 更复杂的对象使用 deepEq 函数进行深度比较
        return deepEq(a, b);
    };
    var toString = Object.prototype.toString
    function deepEq(a,b){
        var className = toString.call(a)
        if(className !== toString.call(b)) return false
        
        switch(className){
            case '[object RegExp]' :
            case '[object String]' :
                return '' + a === '' + b
            case '[object Number]' :
                if(+a !== +a) return +b !== +b
                return +a ===0 ? 1 / +a === 1  / b : +a === +b
            case '[object Date]':
            case '[object Boolean]':
                return +a === +b
        }
    }

    // 第一版 (来自《JavaScript权威指南》)
    function memoize(f) {
        var cache = {};
        return function(){
            var key = arguments.length + Array.prototype.join.call(arguments, ",");
            if (key in cache) {
                return cache[key]
            }
            else {
                return cache[key] = f.apply(this, arguments)
            }
        }
    }
    var add = function(a, b, c) {
        return a + b + c
    }

    var memoizedAdd = memoize(add)

    console.time('use memoize')
    for(var i = 0; i < 100000; i++) {
        memoizedAdd(1, 2, 3)
    }
    console.timeEnd('use memoize')

    console.time('not use memoize')
    for(var i = 0; i < 100000; i++) {
        add(1, 2, 3)
    }
    console.timeEnd('not use memoize')

递归

递归需具备边界条件、递归前进段和递归返回段，当边界条件不满足时，递归前进，当边界条件满足时，递归返回。阶乘中的 n == 1 和 斐波那契数列中的 n < 2 都是边界条件。

阶乘

    function factorial(n) {
        if (n == 1) return n;
        return n * factorial(n - 1)
    }
    console.log(factorial(5)) // 5 * 4 * 3 * 2 * 1 = 120

TIP

引入尾递归概念，就是尾调用自身，一个函数的执行，会创即一个执行上下文，然后压栈进入，执行完毕，弹出栈，没有优化的递归，不断入栈，会造成内存暴增
